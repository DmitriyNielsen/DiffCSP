def process_cif_files(directory):
    data = []
    index = 0  # to simulate 'Unnamed: 0.1' and 'Unnamed: 0'
    for filename in os.listdir(directory):
        if filename.endswith(".cif"):
            filepath = os.path.join(directory, filename)
            try:
                structure = Structure.from_file(filepath)
                formula = structure.formula
                # Generating pseudo thermal properties and bandgap values
                # These should be replaced with actual calculated or sourced values
                heat_all = 1.0  # Placeholder
                heat_ref = 1.0  # Placeholder
                dir_gap = 0.0  # Placeholder
                ind_gap = 0.0  # Placeholder
                space_group_info = structure.get_space_group_info()

                data.append({
                    "Unnamed: 0.1": index,
                    "Unnamed: 0": index,
                    "material_id": index + 1000,  # Simulated material_id
                    "cif": f"# generated using pymatgen\ndata_{formula}\n_symmetry...\nmore CIF data",
                    "formula": formula,
                    "heat_all": heat_all,
                    "heat_ref": heat_ref,
                    "dir_gap": dir_gap,
                    "ind_gap": ind_gap,
                    "spacegroup.number": space_group_info[1],
                    "spacegroup.number.conv": space_group_info[1],
                    "cif.conv": f"# generated using pymatgen\ndata_{formula}\n_symmetry...\nmore CIF data"
                })
                index += 1
            except Exception as e:
                print(f"Failed to process {filename}: {e}")
    
    df = pd.DataFrame(data)
    return df
